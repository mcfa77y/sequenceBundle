48      private LinkedList list = new LinkedList();
49      private String[] headers;

51      public Csv(CsvReader reader) throws IOException {
52          this.headers = reader.readLine();
53  
54          String[] line = null;
55  
56          // suck in the file.
57          // TODO: What's the optimal structure for this
58          while( (line = reader.readLine()) != null) {
59              list.add(line);
60          }
61      }

63      // get all of the unique fields for this header
64      public String[] getAll(String header) {
65          HashSet values = new HashSet();
66          int idx = getHeaderIndex(header);
67          Iterator iterator = list.iterator();
68          while(iterator.hasNext()) {
69              values.add( ((String[])iterator.next())[idx]);
70          }
71          return (String[])values.toArray(new String[0]);
72      }
73  
74      private int getHeaderIndex(String header) {
75          for(int i=0;i<this.headers.length;i++) {
76              if(header.equals(this.headers[i])) {
77                  return i;
78              }
79          }
80          return -1;
81      }

83      public String[] get(String header, String subheader, String value) {
84          HashSet values = new HashSet();
85          int idx = getHeaderIndex(header);
86          int subidx = getHeaderIndex(subheader);
87          Iterator iterator = list.iterator();
88          while(iterator.hasNext()) {
89              String[] strs = (String[])iterator.next();
90              if(value.equals(strs[idx])) {
91                  values.add( strs[subidx] );
92              }
93          }
94          return (String[])values.toArray(new String[0]);
95      }
