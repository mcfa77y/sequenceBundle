48  public class CsvReader {
49  
50      // new String to stop internning
51      static public String END_OF_LINE = new String("END_OF_LINE");
52  
53      private char field_delim = Csv.FIELD_DELIMITER;
54      private char block_delim = Csv.BLOCK_DELIMITER;
55  
56      private Reader reader;
57      private boolean newline;
58  
59      // should   bbb,,,ccc be considered to be two elements?
60      // useful for log parsing.
61      private boolean consume;
62  
63      public CsvReader(Reader rdr) {
64          this.reader = rdr;
65      }
66  
67      public void setFieldDelimiter(char ch) {
68          field_delim = ch;
69      }
70  
71      public void setBlockDelimiter(char ch) {
72          block_delim = ch;
73      }
74  
75      public void setConsuming(boolean b) {
76          this.consume = b;
77      }
78  
79      public boolean isConsuming() {
80          return this.consume;
81      }
82  
83      public String[] readLine() throws IOException {
84          ArrayList list = new ArrayList();
85          String str;
86  
87          while( true ) {
88              str = readField();
89              if(str == null) {
90                  break;
91              }
92              if(str == END_OF_LINE) {
93                  break;
94              }
95              list.add(str);
96          }
97  
98          if(list.isEmpty()) {
99              return null;
100         }
101 
102         return (String[])list.toArray(new String[0]);
103     }
104 
105     public String readField() throws IOException {
106         if(this.newline) {
107             this.newline = false;
108             return END_OF_LINE;
109         }
110 
111         StringBuffer buffer = new StringBuffer();
112         boolean quoted = false;
113         int last = -1;
114         int ch = this.reader.read();
115 
116         if(ch == -1) {
117             return null;
118         }
119 
120         if(ch == '"') {
121             quoted = true;
122         } else 
123         if(ch == block_delim) {
124             return END_OF_LINE;
125         } else
126         if(ch == field_delim) {
127             return "";
128         } else {
129             buffer.append((char)ch);
130         }
131 
132         while( (ch = this.reader.read()) != -1) {
133             if(ch == block_delim) {
134                 this.newline = true;
135                 break;
136             }
137             if(quoted) {
138                 if(ch == '"') {
139                     if(last == '"') {
140                         // forget about this quote and move on
141                         last = -1;  
142                         buffer.append('"');
143                         continue;
144                     }
145                     last = '"';
146                     continue;
147                 }
148             }
149             if(ch == field_delim) {
150                 if(quoted) {
151                     if(last == '"') {
152                         break;
153                     }
154                 } else {
155                     break;
156                 }
157             }
158             buffer.append((char)ch);
159         }
160 
161         return buffer.toString();
162     }
163 
164     public void close() throws IOException {
165         this.reader.close();
166     }
