33  package com.generationjava.io;
34  
35  import java.io.IOException;
36  import java.io.Reader;
37  
38  public class CsvFieldReader extends CsvReader {
39  
40      private String[] headers;
41      private String[] currentLine;
42  
43      public CsvFieldReader(Reader reader) {
44          super(reader);
45      }
46  
47      public void loadHeaders() throws IOException {
48          if(this.headers == null) {
49              useHeaders(super.readLine());
50          }
51      }
52  
53      public void useHeaders(String[] headers) {
54          this.headers = headers;
55      }
56  
57      public String readField(String name) throws IOException {
58          if(this.headers == null) {
59              loadHeaders();
60          }
61  
62          if(this.currentLine == null) {
63              nextBlock();
64              if(this.currentLine == null) {
65                  return null;
66              }
67          }
68  
69          int idx = -1;
70          for(int i=0;i<this.headers.length;i++) {
71              if(name.equals(this.headers[i])) {
72                  idx = i;
73              }
74          }
75  
76          String field = null;
77  
78          if(idx != -1) {
79              field = currentLine[idx];
80          }
81  
82          return field;
83      }
84  
85      public boolean nextBlock() throws IOException {
86          this.currentLine = super.readLine();
87          return (this.currentLine != null);
88      }
89  
90  }
